import spacy

nlp = spacy.load("en_core_web_sm")

def extract_location(text):
    doc = nlp(text)

    for ent in doc.ents:
        if ent.label_ == "GPE":  # Countries, cities
            return ent.text

    return None


def classify_event(text):
    text = text.lower()

    if any(x in text for x in ["war","attack","missile","strike","explosion"]):
        return "critical"

    if any(x in text for x in ["president","election","government"]):
        return "political"

    if any(x in text for x in ["economy","market","inflation"]):
        return "economic"

    return "info"
